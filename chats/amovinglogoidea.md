"use client";
import { useEffect, useRef } from "react";

export default function NadaShaderLogo() {
  const canvasRef = useRef(null);

  useEffect(() => {
    const canvas = canvasRef.current;
    const gl = canvas.getContext("webgl");

    const resize = () => {
      canvas.width = canvas.clientWidth;
      canvas.height = canvas.clientHeight;
      gl.viewport(0, 0, canvas.width, canvas.height);
    };
    window.addEventListener("resize", resize);
    resize();

    // Vertex shader
    const vertex = `
      attribute vec2 position;
      void main() {
        gl_Position = vec4(position, 0.0, 1.0);
      }
    `;

    // Fragment shader (THE CORE)
    const fragment = `
      precision highp float;

      uniform vec2 u_res;
      uniform float u_time;

      // wave source
      float wave(vec2 p, vec2 center) {
        float d = length(p - center);
        return sin(10.0 * d - u_time * 2.0) / (1.0 + d * 4.0);
      }

      void main() {
        vec2 uv = gl_FragCoord.xy / u_res.xy;
        uv = uv * 2.0 - 1.0;
        uv.x *= u_res.x / u_res.y;

        // define sources
        vec2 s1 = vec2(-0.3, 0.0);
        vec2 s2 = vec2(0.3, 0.0);

        float field = 0.0;
        field += wave(uv, s1);
        field += wave(uv, s2);

        // normalize
        float intensity = smoothstep(-0.2, 0.6, field);

        // full spectrum color
        float hue = field * 0.5 + u_time * 0.05;

        vec3 color = 0.5 + 0.5 * cos(6.28318 * (hue + vec3(0.0, 0.33, 0.67)));

        color *= intensity;

        gl_FragColor = vec4(color, 1.0);
      }
    `;

    function compile(type, source) {
      const shader = gl.createShader(type);
      gl.shaderSource(shader, source);
      gl.compileShader(shader);
      return shader;
    }

    const program = gl.createProgram();
    gl.attachShader(program, compile(gl.VERTEX_SHADER, vertex));
    gl.attachShader(program, compile(gl.FRAGMENT_SHADER, fragment));
    gl.linkProgram(program);
    gl.useProgram(program);

    const buffer = gl.createBuffer();
    gl.bindBuffer(gl.ARRAY_BUFFER, buffer);
    gl.bufferData(
      gl.ARRAY_BUFFER,
      new Float32Array([
        -1, -1,
         1, -1,
        -1,  1,
         1,  1,
      ]),
      gl.STATIC_DRAW
    );

    const position = gl.getAttribLocation(program, "position");
    gl.enableVertexAttribArray(position);
    gl.vertexAttribPointer(position, 2, gl.FLOAT, false, 0, 0);

    const u_res = gl.getUniformLocation(program, "u_res");
    const u_time = gl.getUniformLocation(program, "u_time");

    let start = performance.now();

    function render() {
      const t = (performance.now() - start) / 1000;

      gl.uniform2f(u_res, canvas.width, canvas.height);
      gl.uniform1f(u_time, t);

      gl.drawArrays(gl.TRIANGLE_STRIP, 0, 4);

      requestAnimationFrame(render);
    }

    render();
  }, []);

  return (
    <canvas
      ref={canvasRef}
      style={{
        width: "300px",
        height: "300px",
        background: "black",
        display: "block",
      }}
    />
  );
}

import { useEffect, useRef } from "react";

export default function NadaField() {
  const canvasRef = useRef(null);

  useEffect(() => {
    const canvas = canvasRef.current;
    const gl = canvas.getContext("webgl");

    const resize = () => {
      canvas.width = canvas.clientWidth;
      canvas.height = canvas.clientHeight;
      gl.viewport(0, 0, canvas.width, canvas.height);
    };
    window.addEventListener("resize", resize);
    resize();

    // Track mouse
    let mouse = { x: 0, y: 0 };
    canvas.addEventListener("mousemove", (e) => {
      const rect = canvas.getBoundingClientRect();
      mouse.x = (e.clientX - rect.left) / rect.width;
      mouse.y = 1.0 - (e.clientY - rect.top) / rect.height;
    });

    // Vertex shader
    const vertex = `
      attribute vec2 position;
      void main() {
        gl_Position = vec4(position, 0.0, 1.0);
      }
    `;

    // Fragment shader (A + B + C combined)
    const fragment = `
      precision highp float;

      uniform vec2 u_res;
      uniform float u_time;
      uniform vec2 u_mouse;

      #define MAX_SOURCES 5

      // configurable sources
      vec2 sources[MAX_SOURCES];

      float cymaticWave(vec2 p, vec2 center) {
        float d = length(p - center);
        float frequency = 12.0;
        float decay = 3.0;

        return sin(frequency * d - u_time * 2.0) * exp(-decay * d);
      }

      void main() {
        vec2 uv = gl_FragCoord.xy / u_res.xy;
        uv = uv * 2.0 - 1.0;
        uv.x *= u_res.x / u_res.y;

        // base sources (B)
        sources[0] = vec2(-0.4, 0.0);
        sources[1] = vec2(0.4, 0.0);
        sources[2] = vec2(0.0, 0.4);
        sources[3] = vec2(0.0, -0.4);

        // user as source (C)
        vec2 mousePos = (u_mouse * 2.0 - 1.0);
        mousePos.x *= u_res.x / u_res.y;

        float field = 0.0;

        for (int i = 0; i < MAX_SOURCES; i++) {
          field += cymaticWave(uv, sources[i]);
        }

        // add user influence
        field += cymaticWave(uv, mousePos) * 1.5;

        // shaping
        float intensity = smoothstep(-0.3, 0.5, field);

        // full spectrum (energy layer)

        vec3 color = 0.5 + 0.5 * cos(6.28318 * (hue + vec3(0.0, 0.33, 0.67)));

        color *= intensity;

        gl_FragColor = vec4(color, 1.0);
      }
    `;

    function compile(type, source) {
      const shader = gl.createShader(type);
      gl.shaderSource(shader, source);
      gl.compileShader(shader);
      return shader;
    }
